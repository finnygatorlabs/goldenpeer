In your API backend (artifacts/api-server/):

import PDFParse from 'pdf-parse';
import fs from 'fs';

async function extractPDFText(filePath: string): Promise<string> {
  const dataBuffer = fs.readFileSync(filePath);
  const data = await PDFParse(dataBuffer);
  return data.text;
}

Step 2: Add Image OCR (5 minutes)
Use OpenAI Vision API to extract text from images:
TypeScript
import OpenAI from 'openai';

async function extractImageText(imagePath: string): Promise<string> {
  const client = new OpenAI({
    apiKey: process.env.BUILT_IN_FORGE_API_KEY
  });

  const imageData = fs.readFileSync(imagePath);
  const base64Image = imageData.toString('base64');

  const response = await client.vision.beta.messages.create({
    model: 'gpt-4-vision',
    messages: [
      {
        role: 'user',
        content: [
          {
            type: 'image',
            source: {
              type: 'base64',
              media_type: 'image/jpeg',
              data: base64Image
            }
          },
          {
            type: 'text',
            text: 'Extract all text from this image. Return only the text, nothing else.'
          }
        ]
      }
    ]
  });

  return response.content[0].type === 'text' ? response.content[0].text : '';
}
