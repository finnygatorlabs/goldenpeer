Quick Overview
You're adding:
•	✅ Automated reminder delivery every minute via Render cron job
•	✅ Email notifications to family members via Resend
•	✅ Reminder history tracking (completion, missed, snoozed status)
•	✅ Family notification preferences (all/missed only/daily summary/none)
•	✅ Timezone-aware scheduling
•	✅ Zero additional cost

Architecture
Render Cron Job (Free)
    ↓ (every minute)
Your Express API: POST /api/scheduler/reminders
    ↓
Check for due reminders in user's timezone
    ↓
Send emails to primary family member via Resend
    ↓
Update reminder_history table with status

import { format, parse, toZonedTime, fromZonedTime } from "date-fns-tz";
 
interface Reminder {
  id: number;
  userId: number;
  type: string;
  title: string;
  time: string; // "HH:mm"
  frequency: string; // "daily" or "weekly"
  daysOfWeek?: string; // "1,3,5"
  active: boolean;
}
 
export function isReminderDue(reminder: Reminder, userTimezone: string): boolean {
  try {
    const now = new Date();
    const userTime = toZonedTime(now, userTimezone);
 
    const [reminderHours, reminderMinutes] = reminder.time.split(":").map(Number);
    const currentHours = userTime.getHours();
    const currentMinutes = userTime.getMinutes();
 
    const timeMatches = currentHours === reminderHours && currentMinutes === reminderMinutes;
 
    if (!timeMatches) return false;
 
    if (reminder.frequency === "daily") return true;
 
    if (reminder.frequency === "weekly") {
      const dayOfWeek = userTime.getDay();
      const daysOfWeek = reminder.daysOfWeek?.split(",").map(Number) || [];
      return daysOfWeek.includes(dayOfWeek);
    }
 
    return false;
  } catch (error) {
    console.error("Error checking if reminder is due:", error);
    return false;
  }
}
 
export function convertToUserTimezone(utcDate: Date, userTimezone: string): string {
  try {
    const zonedDate = toZonedTime(utcDate, userTimezone);
    return format(zonedDate, "yyyy-MM-dd HH:mm:ss zzz", { timeZone: userTimezone });
  } catch (error) {
    return utcDate.toISOString();
  }
}
 
export function formatReminderTime(time: string): string {
  try {
    const [hours, minutes] = time.split(":").map(Number);
    const period = hours >= 12 ? "PM" : "AM";
    const displayHours = hours > 12 ? hours - 12 : hours === 0 ? 12 : hours;
    return `${displayHours}:${minutes.toString().padStart(2, "0")} ${period}`;
  } catch (error) {
    return time;
  }
}
 
export function isValidTimezone(timezone: string): boolean {
  try {
    Intl.DateTimeFormat(undefined, { timeZone: timezone });
    return true;
  } catch (error) {
    return false;
  }
}

import { Resend } from "resend";
 
const resend = new Resend(process.env.RESEND_API_KEY);
 
interface ReminderNotificationEmailParams {
  familyEmail: string;
  familyName: string;
  reminderType: string;
  reminderTitle: string;
  reminderTime: string;
  notificationPreference: string;
}
 
export async function sendReminderNotificationEmail(
  params: ReminderNotificationEmailParams
): Promise<boolean> {
  try {
    const { familyEmail, familyName, reminderType, reminderTitle, reminderTime, notificationPreference } = params;
 
    if (notificationPreference === "none") {
      console.log(`[Email] Skipping notification for ${familyEmail} (preference: none)`);
      return false;
    }
 
    const reminderTypeDisplay = reminderType
      .split("_")
      .map((word) => word.charAt(0).toUpperCase() + word.slice(1))
      .join(" ");
 
    const [hours, minutes] = reminderTime.split(":").map(Number);
    const period = hours >= 12 ? "PM" : "AM";
    const displayHours = hours > 12 ? hours - 12 : hours === 0 ? 12 : hours;
    const formattedTime = `${displayHours}:${minutes.toString().padStart(2, "0")} ${period}`;
 
    const subject = `Reminder: ${reminderTitle} at ${formattedTime}`;
    const htmlContent = `
      <div style="font-family: Arial, sans-serif; max-width: 600px; margin: 0 auto;">
        <h2 style="color: #333;">Reminder Alert</h2>
        <p>Hi ${familyName},</p>
        <p>Your loved one has a reminder coming up:</p>
        
        <div style="background-color: #f5f5f5; padding: 20px; border-radius: 8px; margin: 20px 0;">
          <p style="margin: 0; color: #666; font-size: 14px;">REMINDER TYPE</p>
          <p style="margin: 10px 0 0 0; font-size: 18px; font-weight: bold; color: #333;">${reminderTypeDisplay}</p>
          
          <p style="margin: 20px 0 0 0; color: #666; font-size: 14px;">REMINDER</p>
          <p style="margin: 10px 0 0 0; font-size: 16px; color: #333;">${reminderTitle}</p>
          
          <p style="margin: 20px 0 0 0; color: #666; font-size: 14px;">SCHEDULED TIME</p>
          <p style="margin: 10px 0 0 0; font-size: 16px; color: #333;">${formattedTime}</p>
        </div>
        
        <p style="color: #999; font-size: 12px; margin-top: 30px;">
          SeniorShield - Helping seniors stay safe and connected
        </p>
      </div>
    `;
 
    const response = await resend.emails.send({
      from: "SeniorShield <noreply@seniorshield.app>",
      to: familyEmail,
      subject: subject,
      html: htmlContent,
    });
 
    if (response.error) {
      console.error(`[Email] Failed to send to ${familyEmail}:`, response.error);
      return false;
    }
 
    console.log(`[Email] Successfully sent to ${familyEmail}`);
    return true;
  } catch (error) {
    console.error("[Email] Error sending reminder notification:", error);
    return false;
  }
}

File 3: artifacts/api-server/src/services/reminder-scheduler.ts
import { db } from "../lib/db";
import { reminders, reminderHistory, familyMembers, users } from "../lib/db/schema";
import { eq, and, lte, gte } from "drizzle-orm";
import { sendReminderNotificationEmail } from "./reminder-email-service";
import { isReminderDue } from "./reminder-utils";
 
interface SchedulerResult {
  processed: number;
  sent: number;
  errors: string[];
}
 
export async function runReminderScheduler(): Promise<SchedulerResult> {
  const result: SchedulerResult = {
    processed: 0,
    sent: 0,
    errors: [],
  };
 
  try {
    console.log("[Scheduler] Starting reminder check...");
 
    const activeReminders = await db
      .select()
      .from(reminders)
      .where(eq(reminders.active, true));
 
    console.log(`[Scheduler] Found ${activeReminders.length} active reminders`);
 
    for (const reminder of activeReminders) {
      try {
        const user = await db
          .select()
          .from(users)
          .where(eq(users.id, reminder.userId))
          .limit(1);
 
        if (!user.length) {
          result.errors.push(`User not found for reminder ${reminder.id}`);
          continue;
        }
 
        const userTimezone = user[0].timezone || "America/New_York";
        result.processed++;
 
        if (isReminderDue(reminder, userTimezone)) {
          const today = new Date();
          today.setHours(0, 0, 0, 0);
 
          const existingHistory = await db
            .select()
            .from(reminderHistory)
            .where(
              and(
                eq(reminderHistory.reminderId, reminder.id),
                eq(reminderHistory.userId, reminder.userId),
                gte(reminderHistory.scheduledTime, today)
              )
            )
            .limit(1);
 
          if (existingHistory.length > 0) {
            continue;
          }
 
          const historyEntry = await db
            .insert(reminderHistory)
            .values({
              reminderId: reminder.id,
              userId: reminder.userId,
              scheduledTime: new Date(),
              sentTime: new Date(),
              status: "sent",
              familyNotified: false,
            })
            .returning();
 
          if (!historyEntry.length) {
            result.errors.push(`Failed to create history for reminder ${reminder.id}`);
            continue;
          }
 
          const primaryFamily = await db
            .select()
            .from(familyMembers)
            .where(
              and(
                eq(familyMembers.userId, reminder.userId),
                eq(familyMembers.isPrimary, true)
              )
            )
            .limit(1);
 
          if (primaryFamily.length > 0) {
            const family = primaryFamily[0];
 
            if (family.notificationPreference !== "none") {
              const emailSent = await sendReminderNotificationEmail({
                familyEmail: family.email,
                familyName: family.name,
                reminderType: reminder.type,
                reminderTitle: reminder.title,
                reminderTime: reminder.time,
                notificationPreference: family.notificationPreference,
              });
 
              if (emailSent) {
                await db
                  .update(reminderHistory)
                  .set({ familyNotified: true })
                  .where(eq(reminderHistory.id, historyEntry[0].id));
 
                result.sent++;
                console.log(`[Scheduler] Sent notification for reminder ${reminder.id}`);
              }
            }
          }
        }
      } catch (error) {
        result.errors.push(
          `Error processing reminder ${reminder.id}: ${error instanceof Error ? error.message : "Unknown error"}`
        );
      }
    }
 
    console.log(
      `[Scheduler] Completed. Processed: ${result.processed}, Sent: ${result.sent}, Errors: ${result.errors.length}`
    );
    return result;
  } catch (error) {
    const errorMsg = error instanceof Error ? error.message : "Unknown error";
    result.errors.push(`Scheduler error: ${errorMsg}`);
    console.error("[Scheduler] Fatal error:", error);
    return result;
  }
}

File 4: artifacts/api-server/src/routes/scheduler.ts
import { Router, Request, Response } from "express";
import { runReminderScheduler } from "../services/reminder-scheduler";
 
const router = Router();
 
router.post("/reminders", async (req: Request, res: Response) => {
  try {
    const schedulerSecret = req.headers["x-scheduler-secret"];
    if (schedulerSecret !== process.env.SCHEDULER_SECRET) {
      return res.status(401).json({ error: "Unauthorized" });
    }
 
    const result = await runReminderScheduler();
 
    return res.json({
      success: true,
      timestamp: new Date().toISOString(),
      processed: result.processed,
      sent: result.sent,
      errors: result.errors,
    });
  } catch (error) {
    console.error("[Scheduler Route] Error:", error);
    return res.status(500).json({
      success: false,
      error: error instanceof Error ? error.message : "Unknown error",
    });
  }
});
 
router.get("/health", (req: Request, res: Response) => {
  return res.json({
    status: "healthy",
    timestamp: new Date().toISOString(),
    uptime: process.uptime(),
  });
});
 
export default router;
